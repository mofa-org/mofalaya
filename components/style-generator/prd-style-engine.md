function normalizeWeights(mix) {
  const sum =
    mix.structure + mix.perception + mix.meaning + mix.distribution || 1;

// Simple segmentation: split by blank lines, fallback to sentence-ish
function segmentText(text) {
  const blocks = text
    .split(/\n\s*\n/g)
    .map((s) => s.trim())
    .filter(Boolean);

  if (blocks.length >= 2) return blocks;

  // fallback: split by punctuation
  const parts = text
    .split(/(?<=[ã€‚ï¼ï¼Ÿ.!?])\s+/g)
    .map((s) => s.trim())
    .filter(Boolean);

  // group into ~4 paragraphs
  const target = 4;
  const grouped = [];
  const chunkSize = Math.max(1, Math.ceil(parts.length / target));
  for (let i = 0; i < parts.length; i += chunkSize) {
    grouped.push(parts.slice(i, i + chunkSize).join(" "));
  }
  return grouped.length ? grouped : [text.trim()].filter(Boolean);
}

    return "development";

      bump("distribution", -0.05);

      bump("structure", -0.03);

    if (role === "opening") {
      bump("structure", 0.06);
      bump("distribution", 0.03);

      bump("perception", 0.06);

      bump("meaning", 0.08);

      bump("meaning", 0.05);
      bump("distribution", 0.03);
